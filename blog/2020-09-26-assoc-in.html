assocIn, or Writing Typed Clojure in Typescript


function assocIn<T, K1 extends keyof T, V extends T[K1]>(
  obj: T,
  path: [K1],
  v: V
): T;

function assocIn<
  T,
  K1 extends keyof T,
  K2 extends keyof T[K1],
  V extends T[K1][K2]
>(obj: T, path: [K1, K2], v: V): T;

function assocIn<
  T,
  K1 extends keyof T,
  K2 extends keyof T[K1],
  K3 extends keyof T[K1][K2],
  V extends T[K1][K2][K3]
>(obj: T, path: [K1, K2, K3], v: V): T;

function assocIn(obj: any, path: any, v: any) {
  const [k1, ...ks] = path;
  return {
    ...obj,
    [k1]: ks.length === 0 ? v : assocIn(obj[k1], ks, v),
  };
}

const x = {
  foo: {
    bar: {
      baz: 1,
      blah: 2,
    },
    nerp: "yar",
  },
  neep: { yip: "yo" },
};

const jlog = (x: any) => console.log(JSON.stringify(x, null, 2));

// Works:
jlog(assocIn(x, ["foo", "nerp"], "Hi!"));

// Works:
jlog(assocIn(x, ["foo", "bar", "baz"], 99));

// Does not typecheck (typo, "buz" should be "baz"):
// jlog(assocIn(x, ["foo", "bar", "buz"], 99));

// Does not typecheck (boolean should be number):
// jlog(assocIn(x, ["foo", "bar", "baz"], true));

function setIn(obj: any, path: any, v: any) {

  return {
